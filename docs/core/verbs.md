We call verb the final method for a given [chain](/core/chain)

## Example

```js
import { fetch } from '@rebased/core';

fetch('kitty', {
  silent: false,
  baseURL: 'https://api.thecatapi.com',
  endpoint: '/v1'
})
  // chain
  .from('http')
  .cache(true)
  .state(false)
  // verb
  .get('/images/search?size=small&mime_types=gif')
  .subscribe(kitty => console.log(kitty));
```
